	\paragraph{0.13}\textbf{Solution:}
		\subparagraph{Claim:}If $P(A) > 0$ 
		\[P(AB | A) \geq P(AB | A \cup B)\]
		\subparagraph{Proof:}
		Since 
		\[P(AB | A) = \frac{P(AB \cap A)}{P(A)}= \frac{P(AB)}{P(A)}\]
		\[P(AB | A \cup B) = \frac{P(AB \cap (A \cup B))}{P(A \cup B)} = \frac{P(AB)}{P(A \cup B)}\]
		So
		\[\frac{P((AB | A)}{P(AB | A \cup B}) = \frac{P(A \cup B)}{P(A)} = \frac{P(A) + P(B) - P(AB)}{P(A)} \geq 1\ \blacksquare\]
		Thus, if $P(A) > 0$ 
		\[P(AB | A) \geq P(AB | A \cup B)\]
